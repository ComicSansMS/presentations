\begin{frame}[fragile]
  \frametitle{Operator overloading (by-value)}
  \begin{lstlisting}
complex operator+(complex x, complex y)
{
    return complex(x.real() + y.real(),
                   x.imag() + y.imag());
}

complex cz = cx + cy;
  \end{lstlisting}
\end{frame}


\begin{frame}[fragile]

}

\end{frame}
